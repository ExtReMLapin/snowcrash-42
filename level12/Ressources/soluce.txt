  $xx =~ tr/a-z/A-Z/;
  $xx =~ s/\s.*//;
   print($xx);
  @output = `egrep "^$xx" /tmp/xd 2>&1`;
  foreach $line (@output) {

      if($s =~ $nn) {

      }
  }
  return 0;
}
